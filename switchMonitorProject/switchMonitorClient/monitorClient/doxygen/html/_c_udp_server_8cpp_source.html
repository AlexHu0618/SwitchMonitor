228 bool CUdpServer::__FrameAnalysis( void )
229 {
230  int nAreaNum = m_ppszDataBuf[0][0]&0xFF;
231  int nAreaFrameNum = 0;
232  int nAreaFrameCounter = 0;
233  for (int nFrameNum = 0; nFrameNum < m_nFrameCounter; nFrameNum++)
234  {
235  if (m_ppszDataBuf[nFrameNum][0]&0xFF == nAreaNum) // all the frames in the same area
236  {
237  nAreaFrameNum = m_ppszDataBuf[0][2]&0xFF | m_ppszDataBuf[0][3]<<8&0xFF00;
238  if (nAreaFrameNum != nAreaFrameCounter)
239  {
240  cout << "miss frame in the air(frmaNum): " << nAreaNum << "(" << nAreaFrameNum << ")" << endl;
241  return false;
242  }
243  nAreaFrameCounter++;
244  }
245  else
246  {
247  nAreaNum = m_ppszDataBuf[nFrameNum][0]&0xFF; //next area
248  nAreaFrameCounter = 0;
249  }
250  }
251  //caculate the start acquiring time
252  int nSecondofAcq = round((int)(m_ppszDataBuf[0][4]&0xFF | m_ppszDataBuf[0][5]<<8&0xFF00 | m_ppszDataBuf[0][6]&0xFF0000 | m_ppszDataBuf[0][7]<<8&0xFF000000)/FREQ);
253  m_tTimeofStartAcq = m_tTimeofFirstRecv;
254  if( m_tTimeofFirstRecv.wSecond < nSecondofAcq )
255  {
256  m_tTimeofStartAcq.wMinute = m_tTimeofFirstRecv.wMinute;
257  m_tTimeofStartAcq.wSecond = m_tTimeofFirstRecv.wSecond + 60 - nSecondofAcq;
258  }
259  else
260  {
261  m_tTimeofStartAcq.wSecond = m_tTimeofFirstRecv.wSecond - nSecondofAcq;
262  }
263  return true;
264 }
